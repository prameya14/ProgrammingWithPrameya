# Programme Topic: Snake, Water and Gun game using Python

import random

i = 1
user_points = 0
comp_points = 0

chance = int(input("Enter how many times do you want to play\n"))

while i <= chance:

    def game_logic(comp, user):
        if comp == user:
            return None
        elif comp == "s":
            if user == "w":
                return False
            elif user == "g":
                return True
        elif comp == "w":
            if user == "s":
                return True
            elif user == "g":
                return False
        elif comp == "g":
            if user == "s":
                return False
            elif user == "w":
                return True

    print("Computer's Turn: Snake(s), Water(w) and Gun(g)\n")
    rand = random.randint(1, 3)

    if rand == 1:
        comp_turn = "s"
    elif rand == 2:
        comp_turn = "w"
    else:
        comp_turn = "g"

    user_turn = input("User's Turn: Snake(s), Water(w) and Gun(g)\n")

    result = game_logic(comp_turn, user_turn)

    print(f"Computer Chose: {comp_turn}")
    print(f"User Chose: {user_turn}")

    # print(result)

    if result == None:
        res = "Draw"
    elif result:
        res = "User Won"
        user_points += 1
    else:
        res = "Computer Won"
        comp_points += 1

    print(f"Result: {res}")

    i += 1

if user_points == comp_points:
    final_res = "Draw"
elif user_points < comp_points:
    final_res = "Computer Won"
else:
    final_res = "User Won"

print(f"Final Results:\nComputer's Points = {comp_points}\nUser's Points = {user_points}\nFinal Result = {final_res}")
